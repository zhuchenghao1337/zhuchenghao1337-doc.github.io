# 动态路由

提示

根据后端数据遍历生成el-menu组件

点击查看 
<template>
  <div>
    <template v-for="item in treeData">
      <el-submenu
        :index="item.id"
        v-if="item.children && item.children.length > 0 && item.children[0].type.toString() === '1'">
        <template slot="title">
          <i :class="item.icon"></i>
          <span>{{ item.name}}</span>
        </template>
        <TreeMenu :tree-data="item.children"></TreeMenu>
      </el-submenu>
      <el-menu-item
        v-else-if="item.type.toString() === '1'"
        :index="item.path"
        :key="item.id"
      >
        <template slot="title">
          <i v-if="item.icon" :class="item.icon"></i>
          <router-link :to="item.path">{{ item.name}}</router-link>
        </template>
      </el-menu-item>
    </template>
  </div>
</template>
